int func fib( f : int)
{
  if(f < 2) {
    return(f);
  } else {
    return( (call fib(f - 1)) + (call fib(f - 2)) );
  }

}

int func factorial( n : int ) {
  if( n == 1 ) {
    return(1);
  } else {
    return(n * ( call factorial(n - 1) ));
  }
}
